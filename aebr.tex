% This is the document class for AEBRs. Note that this format is interchangable with
% report. Any AEBR will compile as a dragonfly-report and vice versa.
% The draft option will also work. 
\documentclass{aebr}

% The AEBR requires two titles. The first is the short title which goes on the 
% page footers. The second is the long title which goes on the cover page. 
% You are only allowed to use the the LaTeX macros \\ and \emph inside of titles
% safely. Others may result in unpredictable behaviour. 
\title{Example}{An Example\\ Report about \emph{sheep} but that also has such
a long title that it actually has to wrap!}

% Each AEBR has a subtitle
\subtitle{A report for testing purposes}

%The date must be split into two parts for compatability with reports. 
% The year is specified with \date, and the month is specified with \reportmonth
\date{2014}
\reportmonth{May}

% Authors must be listed with their full names separated by \and.
\author{John Smith \and Jane E. Smith \and Joe Blogs}

% The subcaption package is used to create subfigures in this example.
\usepackage{subcaption}

% The lipsum package is used to generate filler text.
\usepackage{lipsum}

% Each report has an ISSN, ISBN and report number. If any of these are not 
% included a place holder value will be used.
\issn{1234-5678}
\isbn{12-34567-89}
\reportno{79}

% This specified the location of the bibliography file
\addbibresource{test.bib}

% The document begins here
\begin{document}

% Generate the title page
\maketitle

% Generate the table of contents. This should be done for AEBRs.
\tableofcontents

% An AEBR must contain an executive summary. Which \summary will generate a 
% title and contents entry for. 

\summary

% The summary must start with a citation for the current document which can be 
% automatically generated using \citeself.
\citeself
