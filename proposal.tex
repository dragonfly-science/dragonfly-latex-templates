% This is an example of a proposal. Like all documents it can be given the draft 
% option to place a watermark, increase linespacing and add in line numbers.
% The proposal is almost identical to the report format, so if there is any
% confusion consult report.tex. 
\documentclass{dragonfly-proposal}

% A proposal requires two titles. The first is the short title. This is used for the 
% footer text. The second is the full title. This goes on the front of the report. 
% You are only allowed to use two LaTeX macros in the title: \\ and \emph. If others
% are used there may be issues.
\title{Example}{An Example\\ Report about \emph{sheep}}

% Proposals require a subtitle which goes below the main title on the cover page. 
\subtitle{A report for testing purposes}

% Authors must be included for a proposal. They should be listed with their full names
% with each author separated by \and.
\author{John Smith \and Jane E. Smith}

% In this example I have used the subcaption package to create subfloats. These are
% when you have more than one picture within a single figure, and each gets it's own
% internal caption. 
\usepackage{subcaption}

% Optionally you can set an image to go on the titlepage. If this is not set it will
% default to the wing image as a banner accross the top of the page. If an image is

% 210x110 mmm. Wider and taller images may not work correctly. If the image is shorter
% than written, the width should not be reduced.
